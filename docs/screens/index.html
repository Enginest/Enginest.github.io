function BottomBar(navigation) {
  return (
    <Tab.Navigator initialRouteName="Home" options={{ labeled: false, }} screenOptions={{ tabBarHideOnKeyboard: true, tabBarShowLabel: false, backgroundColor: global.params[0], tabBarActiveTintColor: 'white', tabBarInactiveTintColor: '#778899', tabBarStyle: { backgroundColor: '#000', height: 70 }, headerShown: false }} >
      <Tab.Screen
        name="Profile"
        component={Profile}
        options={{
          tabBarLabel: 'Profile',
          tabBarIcon: ({ color, size }) => (
            <MaterialCommunityIcons name="account" color={color} size={size} />
          ),
        }}
      />

      <Tab.Screen
        name="Home"
        component={Home}
        options={{
          tabBarLabel: 'Home',
          tabBarIcon: ({ color, size }) => (
            <MaterialCommunityIcons name="home" color={color} size={size} />
          ),
        }}
      />

      <Tab.Screen
        name="Database"
        component={Database}
        options={{
          headerTintColor: '#',
          tabBarLabel: 'Database',
          tabBarIcon: ({ color, size, focused }) => {
            return (
              <MaterialCommunityIcons name="database" color={color} size={size} />
            )
          },

        }}

      />

    </Tab.Navigator>
  );
}

Upload​

Fonctionne pas. C'est la fonction de test.
Le serveur n'autorise pas la methode POST. A voir.

POEM_Mobile/screen/Programs.js
const uploadFile = async () => {
      try {
        const document = await DocumentPicker.getDocumentAsync({
          type: '*/*',
        });
    
        if (document.type === 'success') {
          const file = document.uri;
          const fileName = document.name;
    
          const formData = new FormData();
          formData.append('file', { uri: file, name: fileName, type: 'application/octet-stream' });
    
          const response = await axios.post('http://10.0.2.2:80/upload', formData, {
            headers: {
              'Content-Type': 'multipart/form-data',
            },
            onUploadProgress: (progressEvent) => {
              const progress = (progressEvent.loaded / progressEvent.total) * 100;
              console.log(`Taux de progression : ${Math.round(progress)}%`);
            },
          });
    
          console.log('Réponse du serveur:', response.data);
        }
      } catch (error) {
        console.error('Erreur lors de l\'upload du fichier:', error);
      }
    };

POEM_Mobile/screen/Programs.js
const downloadFile = async () => {
        try {
            const fileUrl = 'http://10.0.2.2:80/download/Test.zip';
            const fileUri = FileSystem.documentDirectory + 'Test.zip';
      
            const downloadObject = FileSystem.createDownloadResumable(
              fileUrl,
              fileUri,
              {},
              (downloadProgress) => {
                const progress = downloadProgress.totalBytesWritten / downloadProgress.totalBytesExpectedToWrite;
                console.log(`Progression du téléchargement : ${progress * 100}%`);
              }
            );
      
            const response = await downloadObject.downloadAsync();
      
            if (response.status === 200) {
              console.log('Fichier téléchargé avec succès !');
              console.log(`Chemin du fichier local : ${fileUri}`);
            } else {
              console.log('Erreur lors du téléchargement du fichier.');
            }
          } catch (error) {
            console.error('Une erreur s\'est produite :', error);
          }
        };
